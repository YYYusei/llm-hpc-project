## Example 1: Stencil Computation (C++)

### Code:

```cpp
// 2D 5-point stencil (heat diffusion)
void stencil_2d(double* u_new, double* u, int nx, int ny, double cx, double cy) {{
    for (int j = 1; j < ny-1; j++) {{
        for (int i = 1; i < nx-1; i++) {{
            int idx = j * nx + i;
            u_new[idx] = u[idx] + cx * (u[idx-1] - 2*u[idx] + u[idx+1])
                                + cy * (u[idx-nx] - 2*u[idx] + u[idx+nx]);
        }}
    }}
}}
```

## Example 2: DFT Plane-Wave Operations (Fortran)

### Code:
```fortran
! Apply nonlocal projectors to wavefunctions
subroutine nonlop(cg, vnl_proj, natom, npw, nband, enl)
  implicit none
  integer, intent(in) :: natom, npw, nband
  complex(dp), intent(in) :: cg(npw, nband)
  complex(dp), intent(in) :: vnl_proj(npw, natom)
  real(dp), intent(out) :: enl(nband)
  
  integer :: iband, iatom, ipw
  complex(dp) :: proj_coef
  
  do iband = 1, nband
    enl(iband) = 0.0_dp
    do iatom = 1, natom
      proj_coef = (0.0_dp, 0.0_dp)
      ! Inner loop over plane waves - main hotspot
      do ipw = 1, npw
        proj_coef = proj_coef + conjg(vnl_proj(ipw, iatom)) * cg(ipw, iband)
      end do
      enl(iband) = enl(iband) + real(proj_coef * conjg(proj_coef))
    end do
  end do
end subroutine
```

## Example 3: 3D FFT Operations (Fortran)

### Code:
```fortran
! 3D FFT for plane-wave DFT (real to complex)
subroutine fourwf(ngfft, npw, ndat, cg, ur)
  implicit none
  integer, intent(in) :: ngfft(3), npw, ndat
  complex(dp), intent(in) :: cg(npw, ndat)
  real(dp), intent(out) :: ur(ngfft(1), ngfft(2), ngfft(3), ndat)
  
  integer :: idat, i1, i2, i3
  complex(dp), allocatable :: work(:,:,:)
  
  allocate(work(ngfft(1), ngfft(2), ngfft(3)))
  
  do idat = 1, ndat
    ! Scatter plane waves to 3D grid
    call pw_to_grid(cg(:,idat), work, npw, ngfft)
    
    ! 3D FFT (main computation)
    call fft_3d(work, ngfft, -1)  ! -1 = backward FFT
    
    ! Extract real part
    do i3 = 1, ngfft(3)
      do i2 = 1, ngfft(2)
        do i1 = 1, ngfft(1)
          ur(i1, i2, i3, idat) = real(work(i1, i2, i3))
        end do
      end do
    end do
  end do
  
  deallocate(work)
end subroutine
```

## Example 4: Sparse Matrix Operations (Fortran)

### Code:
```fortran
! Sparse matrix-vector multiply in CSR format
subroutine spmv_csr(nrows, rowptr, colind, values, x, y)
  implicit none
  integer, intent(in) :: nrows
  integer, intent(in) :: rowptr(nrows+1), colind(*)
  real(dp), intent(in) :: values(*), x(*)
  real(dp), intent(out) :: y(nrows)
  
  integer :: i, j, jstart, jend
  real(dp) :: sum
  
  do i = 1, nrows
    sum = 0.0_dp
    jstart = rowptr(i)
    jend = rowptr(i+1) - 1
    do j = jstart, jend
      sum = sum + values(j) * x(colind(j))
    end do
    y(i) = sum
  end do
end subroutine
```

## Analysis Guidelines

When analyzing code, consider:

1. **Language recognition**: 
   - C/C++: `for` loops, pointers, `->` operator
   - Fortran: `do` loops, array slicing, `intent` attributes, column-major arrays

2. **Hotspot identification**: Look for nested loops, expensive operations (division, transcendentals, complex arithmetic), and high iteration counts

3. **Bottleneck classification**: 
   - Compute-bound: High FLOPs, optimized memory access, arithmetic intensity > 10
   - Memory-bound: Indirect indexing, irregular access, arithmetic intensity < 1
   - Mixed: 1-10 FLOPs/byte

4. **DFT/Quantum chemistry patterns**:
   - FFT operations → usually memory-bound
   - Projector operations → usually compute-bound  
   - Matrix diagonalization → compute-bound (BLAS/LAPACK)
   - Exchange-correlation → compute-bound

5. **GPU suitability**: Consider parallelism, memory patterns, and data transfer overhead
