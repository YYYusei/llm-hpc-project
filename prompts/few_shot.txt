## Example 3: Particle Pairwise Interaction (N-body style with neighbor list)

### Code:
```cpp
// Compute pairwise forces with cutoff and neighbor list
void compute_forces(int n, double* x, double* y, double* z,
                    double* fx, double* fy, double* fz,
                    int* neighbors, int* num_neighbors, double cutoff_sq) {{
    for (int i = 0; i < n; i++) {{
        double fxi = 0.0, fyi = 0.0, fzi = 0.0;
        double xi = x[i], yi = y[i], zi = z[i];
        
        for (int k = 0; k < num_neighbors[i]; k++) {{
            int j = neighbors[i * MAX_NEIGHBORS + k];
            double dx = xi - x[j];
            double dy = yi - y[j];
            double dz = zi - z[j];
            double rsq = dx*dx + dy*dy + dz*dz;
            
            if (rsq < cutoff_sq && rsq > 0.0) {{
                double r2inv = 1.0 / rsq;
                double r6inv = r2inv * r2inv * r2inv;
                double force = 48.0 * r6inv * (r6inv - 0.5) * r2inv;
                fxi += dx * force;
                fyi += dy * force;
                fzi += dz * force;
            }}
        }}
        fx[i] += fxi;
        fy[i] += fyi;
        fz[i] += fzi;
    }}
}}
```
